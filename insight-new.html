Supabase RLS SQL — Copy & Deploy
-- Enable RLS ALTER TABLE insights ENABLE ROW LEVEL SECURITY; ALTER TABLE form_responses ENABLE ROW LEVEL SECURITY; -- Own data only CREATE POLICY "client_own" ON insights FOR SELECT USING ( client_id = auth.uid() ); -- Form responses: owner sees all, client sees own CREATE POLICY "resp_owner" ON form_responses FOR SELECT USING ( is_owner(auth.uid()) OR session_key = current_setting('app.session', true) ); -- Public insert (no auth needed for form fills) CREATE POLICY "resp_insert" ON form_responses FOR INSERT WITH CHECK (true);
