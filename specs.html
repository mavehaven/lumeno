Installation

npm install @lumenos/sdk

Quick Start

import { LumenOS } from '@lumenos/sdk';

// Initialize the SDK
const mirror = new LumenOS({
    apiKey: 'your-api-key',
    widgets: ['calendar', 'weather']
});

// Start the mirror
await mirror.initialize();

// Add a widget
await mirror.widgets.add({
    type: 'weather',
    position: { x: 0, y: 0 },
    settings: {
        location: 'New York',
        unit: 'celsius'
    }
});
